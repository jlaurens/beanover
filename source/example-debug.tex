\cs_new:Npn \MyDoExpect:Nn #1 #2 {
  \BeanovesEval[\MyExpect_tl]{#2}
  \int_compare:nNnTF {\insertslidenumber} #1 {\MyExpect_tl} {\color{Green}(}{\color{Red}(FAILURE:~}
}
\cs_generate_variant:Nn \MyDoExpect:Nn { Nx }
\cs_new:Npn \MyExpect:Nn #1#2 {
  \group_begin:
  \MyDoExpect:Nx #1 {#2}
  \insertslidenumber \ensuremath{#1} #2)
  \group_end:
}

  \MyExpect:Nn =

  \MyExpect:Nn < { #1 + 1 }
}

  \MyExpect:Nn > { #1 - 1 }
}
\cs_new:Npn \MyExpectInRange #1 {
  \group_begin:
  \tl_clear:N \l__bnvs_a_tl
  \tl_clear:N \l__bnvs_b_tl
  \tl_clear:N \l__bnvs_c_tl
  \regex_split:NnN \c__bnvs_colons_regex { #1 } \l__bnvs_split_seq
  \seq_pop_left:NNT \l__bnvs_split_seq \l__bnvs_a_tl {
    \seq_pop_left:NNT \l__bnvs_split_seq \l__bnvs_b_tl {
      \tl_if_empty:NTF \l__bnvs_b_tl {
        \seq_pop_left:NN \l__bnvs_split_seq \l__bnvs_b_tl
        \seq_pop_left:NNT \l__bnvs_split_seq \l__bnvs_c_tl {
          \tl_if_empty:NTF \l__bnvs_c_tl {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(1):~#1 }
          } {
            \int_compare:nNnT { \tl_count:N \l__bnvs_c_tl } > { 1 } {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(2):~#1 }
            }
            \seq_pop_left:NN \l__bnvs_split_seq \l__bnvs_c_tl
            \seq_if_empty:NF \l__bnvs_split_seq {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(3):~#1 }
            }
          }
        }
      } {
        \int_compare:nNnT { \tl_count:N \l__bnvs_b_tl } > { 1 } {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(4):~#1 }
        }
        \seq_pop_left:NN \l__bnvs_split_seq \l__bnvs_c_tl
        \seq_pop_left:NNTF \l__bnvs_split_seq \l__bnvs_b_tl {
          \tl_if_empty:NTF \l__bnvs_b_tl {
            \seq_pop_left:NN \l__bnvs_split_seq \l__bnvs_b_tl
            \seq_if_empty:NF \l__bnvs_split_seq {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(5):~#1 }
            }
          } {
\msg_error:nnn { beanoves } { :n } { Invalid~range~expression(6):~#1 }
          }
        } {
          \tl_clear:N \l__bnvs_b_tl
        }
      }
    }
  }
  \tl_if_empty:NTF \l__bnvs_a_tl {
    \tl_if_empty:NTF \l__bnvs_b_tl {
      \tl_if_empty:NTF \l__bnvs_c_tl {
      } {
      }
    } {
      \tl_if_empty:NTF \l__bnvs_c_tl {
      } {
      }
    }
  } {
    \tl_if_empty:NTF \l__bnvs_b_tl {
      \tl_if_empty:NTF \l__bnvs_c_tl {
      } {
      }
    } {
      \tl_if_empty:NTF \l__bnvs_c_tl {
      } {
      }
    }
  }
  \group_end:
}
